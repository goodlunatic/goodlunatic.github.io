using namespace std;

int main() {
	int n;cin >> n;
	int cnt = 0;
	while (n > 1) {
		if (n % 2 == 1)n = (3 * n + 1) / 2;
		else n /= 2;
		cnt++;
	}
	cout<<cnt<<endl;
	return 0;
}

#include <bits/stdc++.h>  
using namespace std;  
  
string table[10] = {"ling", "yi", "er", "san", "si", "wu", "liu", "qi", "ba", "jiu"};  
vector<int>v;  
int main() {  
//    freopen("input.txt", "r", stdin);  
    long long sum = 0;  
    string s;  
    cin >> s;  
    long long len = s.length();  
    for (int i = 0; i < len; i++) {  
        sum += int(s[i]) - int('0') ;  
    }  
    while (sum) {  
        int tmp = sum % 10;  
        v.push_back(tmp);  
        sum /= 10;  
    }  
    for (int i = v.size() - 1; i > 0; i--) {  
        cout << table[v[i]] << ' ';  
    }  
    cout << table[v[0]];  
    return 0;  
}

#include <bits/stdc++.h>  
using namespace std;  
  
int main() {  
//    freopen("input.txt", "r", stdin);  
    int n;  
    cin >> n;  
    string s;  
    while (n--) {  
        cin >> s;  
        map<char, int> m;  
        int l=0, r=0;  
        int len = s.length();  
        for (int i = 0; i < len; i++) {  
            m[s[i]]++;  
            if (s[i] == 'P') l = i;  
            if (s[i] == 'T') r = i;  
        }  
        if (m['P'] == 1 && m['A'] != 0 && m['T'] == 1 && m.size() == 3 && (r - l) != 1 && l * (r - l - 1) == (len - r - 1)) {  
            cout << "YES" << endl;  
        } else cout << "NO" << endl;  
    }  
    return 0;  
}
